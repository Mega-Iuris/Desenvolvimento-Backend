CREATE DATABASE db_escola_tech;

USE db_escola_tech;

CREATE TABLE tb_enderecos(
  id INT PRIMARY KEY AUTO_INCREMENT,
  logradouro VARCHAR(255),
  bairro VARCHAR(255),
  numero INT,
  uf CHAR(2),
  pais VARCHAR(255)
);

CREATE TABLE tb_filial (
  id INT PRIMARY KEY AUTO_INCREMENT,
  nome VARCHAR(255),
  cnpj CHAR(14),
  id_endereco INT,
  FOREIGN KEY (id_endereco) REFERENCES tb_enderecos(id)
);
